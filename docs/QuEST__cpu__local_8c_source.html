116 qreal densmatr_calcTotalProb(Qureg qureg) {
117 
118  // computes the trace using Kahan summation
119  qreal pTotal=0;
120  qreal y, t, c;
121  c = 0;
122 
123  long long int numCols = 1LL << qureg.numQubitsRepresented;
124  long long diagIndex;
125 
126  for (int col=0; col< numCols; col++) {
127  diagIndex = col*(numCols + 1);
128  y = qureg.stateVec.real[diagIndex] - c;
129  t = pTotal + y;
130  c = ( t - pTotal ) - y; // brackets are important
131  pTotal = t;
132  }
133 
134  // does not check imaginary component, by design
135 
136  return pTotal;
137 }

139 qreal statevec_calcTotalProb(Qureg qureg){
140  // implemented using Kahan summation for greater accuracy at a slight floating
141  // point operation overhead. For more details see https://en.wikipedia.org/wiki/Kahan_summation_algorithm
142  qreal pTotal=0;
143  qreal y, t, c;
144  long long int index;
145  long long int numAmpsPerRank = qureg.numAmpsPerChunk;
146  c = 0.0;
147  for (index=0; index<numAmpsPerRank; index++){
148  // Perform pTotal+=qureg.stateVec.real[index]*qureg.stateVec.real[index]; by Kahan
149 
150  y = qureg.stateVec.real[index]*qureg.stateVec.real[index] - c;
151  t = pTotal + y;
152  // Don't change the bracketing on the following line
153  c = ( t - pTotal ) - y;
154  pTotal = t;
155 
156  // Perform pTotal+=qureg.stateVec.imag[index]*qureg.stateVec.imag[index]; by Kahan
157 
158  y = qureg.stateVec.imag[index]*qureg.stateVec.imag[index] - c;
159  t = pTotal + y;
160  // Don't change the bracketing on the following line
161  c = ( t - pTotal ) - y;
162  pTotal = t;
163 
164 
165  }
166  return pTotal;
167 }
