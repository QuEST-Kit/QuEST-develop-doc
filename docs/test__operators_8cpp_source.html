418 TEST_CASE( "applyPauliSum", "[operators]" ) {
419 
420  Qureg vecIn = createQureg(NUM_QUBITS, QUEST_ENV);
421  Qureg vecOut = createQureg(NUM_QUBITS, QUEST_ENV);
422  Qureg matIn = createDensityQureg(NUM_QUBITS, QUEST_ENV);
423  Qureg matOut = createDensityQureg(NUM_QUBITS, QUEST_ENV);
424 
425  initDebugState(vecIn);
426  initDebugState(matIn);
427 
428  SECTION( "correctness" ) {
429 
430  int numTerms = GENERATE( 1, 2, 10, 15);
431  int numPaulis = numTerms * NUM_QUBITS;
432 
433  // each test will use random coefficients
434  qreal coeffs[numTerms];
435  for (int i=0; i<numTerms; i++)
436  coeffs[i] = getRandomReal(-5, 5);
437 
438  /* it's too expensive to try ALL Pauli sequences, via
439  * pauliOpType* paulis = GENERATE_COPY( pauliseqs(numPaulis) );.
440  * Furthermore, take(10, pauliseqs(numTargs)) will try the same pauli codes.
441  * Hence, we instead opt to repeatedly randomly generate pauliseqs
442  */
443  GENERATE( range(0,10) ); // gen 10 random pauli-codes
444  pauliOpType paulis[numPaulis];
445  for (int i=0; i<numPaulis; i++)
446  paulis[i] = (pauliOpType) getRandomInt(0,4);
447 
448  // build the resulting Pauli sum; sum_j coeffs[j] * paulis[n*j : (n+1)j]
449  QMatrix iMatr{{1,0},{0,1}};
450  QMatrix xMatr{{0,1},{1,0}};
451  QMatrix yMatr{{0,-1i},{1i,0}};
452  QMatrix zMatr{{1,0},{0,-1}};
453  QMatrix pauliSum = getZeroMatrix(1<<NUM_QUBITS);
454 
455  for (int t=0; t<numTerms; t++) {
456  QMatrix pauliProd = QMatrix{{1}};
457 
458  for (int q=0; q<NUM_QUBITS; q++) {
459  int i = q + t*NUM_QUBITS;
460 
461  QMatrix fac;
462  if (paulis[i] == PAULI_I) fac = iMatr;
463  if (paulis[i] == PAULI_X) fac = xMatr;
464  if (paulis[i] == PAULI_Y) fac = yMatr;
465  if (paulis[i] == PAULI_Z) fac = zMatr;
466  pauliProd = getKroneckerProduct(fac, pauliProd);
467  }
468  pauliSum += coeffs[t] * pauliProd;
469  }
470 
471  SECTION( "state-vector" ) {
472 
473  QVector vecRef = toQVector(vecIn);
474  applyPauliSum(vecIn, paulis, coeffs, numTerms, vecOut);
475 
476  // ensure vecIn barely changes under precision
477  REQUIRE( areEqual(vecIn, vecRef) );
478 
479  // ensure vecOut changed correctly
480  REQUIRE( areEqual(vecOut, pauliSum * vecRef) );
481  }
482  SECTION( "density-matrix" ) {
483 
484  QMatrix matRef = toQMatrix(matIn);
485  applyPauliSum(matIn, paulis, coeffs, numTerms, matOut);
486 
487  // ensure matIn barely changes under precision
488  REQUIRE( areEqual(matIn, matRef) );
489 
490  // ensure matOut changed correctly
491  REQUIRE( areEqual(matOut, pauliSum * matRef, 1E2*REAL_EPS) );
492  }
493  }
