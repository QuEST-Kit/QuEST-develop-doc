123  SECTION( "correctness" ) {
124 
125  int numTargs = GENERATE( range(1,NUM_QUBITS+1) );

128  /* it's too expensive to try ALL Pauli sequences, via

139  QMatrix iMatr{{1,0},{0,1}};
