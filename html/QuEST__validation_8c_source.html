143 #define macro_isMatrixUnitary(m, dim, retVal) { \
144  /* elemRe_ and elemIm_ must not exist in caller scope */ \
145  qreal elemRe_, elemIm_; \
146  retVal = 1; \
147  /* check m * ConjugateTranspose(m) == Identity */ \
148  for (int r=0; r < (dim); r++) { \
149  for (int c=0; c < (dim); c++) { \
150  /* m[r][...] * ConjugateTranspose(m)[...][c] */ \
151  elemRe_ = 0; \
152  elemIm_ = 0; \
153  for (int i=0; i < (dim); i++) { \
154  /* m[r][i] * conj(m[c][i]) */ \
155  elemRe_ += m.real[r][i]*m.real[c][i] + m.imag[r][i]*m.imag[c][i]; \
156  elemIm_ += m.imag[r][i]*m.real[c][i] - m.real[r][i]*m.imag[c][i]; \
157  } \
158  /* check distance from identity */ \
159  if ((absReal(elemIm_) > REAL_EPS) || \
160  (r == c && absReal(elemRe_ - 1) > REAL_EPS) || \
161  (r != c && absReal(elemRe_ ) > REAL_EPS)) { \
162  retVal = 0; \
163  break; \
164  } \
165  } \
166  if (retVal == 0) \
167  break; \
168  } \
169 }

189 #define macro_isCompletelyPositiveMap(ops, numOps, opDim, retVal) { \
190  /* dist_, elemRe_, elemIm_ and difRe_ must not exist in caller scope */ \
191  qreal dist_ = 0; \
192  qreal elemRe_, elemIm_, difRe_; \
193  for (int n=0; n<(numOps); n++) \
194  /* calculate Hilbert schmidt distance of ConjugateTranspose(op)*op to Identity matrix */ \
195  for (int r=0; r<(opDim); r++) \
196  for (int c=0; c<(opDim); c++) { \
197  /* elem = ConjugateTranspose(op)[r][...] (dot) op[...][c] */ \
198  elemRe_ = 0; \
199  elemIm_ = 0; \
200  for (int i=0; i<(opDim); i++) { \
201  elemRe_ += ops[n].real[i][r]*ops[n].real[i][c] + ops[n].imag[i][r]*ops[n].imag[i][c]; \
202  elemIm_ += ops[n].real[i][r]*ops[n].imag[i][c] - ops[n].imag[i][r]*ops[n].real[i][c]; \
203  } \
204  /* Hilbert schmidt distance of element to that of identity matrix */ \
205  difRe_ = elemRe_ - ((r==c)? 1:0); \
206  dist_ += difRe_*difRe_ + elemIm_*elemIm_; \
207  } \
208  retVal = (dist_ < REAL_EPS); \
209 }

229 int areUniqueQubits(int* qubits, int numQubits) {
230  long long int mask = 0;
231  long long int bit;
232  for (int q=0; q < numQubits; q++) {
233  bit = 1LL << qubits[q];
234  if (mask & bit)
235  return 0;
236  mask |= bit;
237  }
238  return 1;
239 }
